Пример с Express:
app.get('/bankAPI/transactions/:search', (req, res) => {
  const search = req.params.search;
  const iban = req.query.iban;

  const query = `
    SELECT * FROM transactions 
    WHERE dateTime LIKE ? 
    AND (IBAN_sender = ? OR IBAN_receiver = ?)
  `;

  connection.query(query, [`%${search}%`, iban, iban], (err, results) => {
    if (err) {
      return res.status(600).json({ error: 'DB error' });
    }

    if (results.length === 0) {
      return res.status(404).json({ message: 'Няма транзакции' });
    }

    res.json(results);
  });
});
🔹 2. POST /bankAPI/transactions/

Това е най-важната част (междубанков превод)

Какво се случва:
🪜 Стъпки:
Проверяваш:
има ли IBAN_sender
има ли баланс
лимит
Взимаш кода на банката:
const bankCode = IBAN_receiver.substring(0, 3);
Викаш банковия регистър:
GET /bankRegister/bank/<IBAN_receiver>
Получаваш API адрес
Правиш заявка към другата банка (axios/fetch)
Ако успех:
записваш транзакцията
намаляваш баланса
Пример (опростен):
import axios from 'axios';

app.post('/bankAPI/transactions', async (req, res) => {
  const { IBAN_sender, IBAN_receiver, amount, currency, reason } = req.body;

  try {
    // 1. Проверка на баланс
    const [account] = await getAccount(IBAN_sender);

    if (!account) {
      return res.status(605).json({ msg: 'Невалиден IBAN' });
    }

    if (account.balance < amount) {
      return res.status(601).json({ msg: 'Недостатъчна наличност' });
    }

    // 2. Вземане на банката
    const bankCode = IBAN_receiver.substring(0, 3);

    const registryRes = await axios.get(
      `http://localhost:4000/bankRegister/bank/${IBAN_receiver}`
    );

    const bankApi = registryRes.data.bank_api;

    // 3. Заявка към другата банка
    const response = await axios.post(bankApi, {
      IBAN_sender,
      IBAN_receiver,
      amount,
      currency,
      reason,
    });

    if (response.data.status_code !== 200) {
      return res.status(650).json(response.data);
    }

    // 4. Запис в DB
    await saveTransaction(...);

    // 5. Update balance
    await updateBalance(...);

    res.json({ msg: 'Успешен превод' });

  } catch (err) {
    return res.status(602).json({ msg: 'Registry error' });
  }
});
🔹 3. Банков регистър (bankRegister API)

Това е най-лесната част.

връща коя е банката
Пример:
app.get('/bankRegister/bank/:iban', (req, res) => {
  const iban = req.params.iban;
  const code = iban.substring(0, 3);

  const query = 'SELECT * FROM banks WHERE code = ?';

  connection.query(query, [code], (err, result) => {
    if (err) {
      return res.status(702).json({ msg: 'DB error' });
    }

    if (result.length === 0) {
      return res.status(701).json({ msg: 'Unknown bank' });
    }

    res.json(result[0]);
  });
});

🟢 1. Вземане на транзакции (GET)
👉 Добави това в App.jsx
import { useState, useEffect } from 'react';

function App() {
  const [transactions, setTransactions] = useState([]);

  useEffect(() => {
    fetch('/api/posts') // или /bankAPI/transactions/...
      .then(res => res.json())
      .then(data => setTransactions(data))
      .catch(err => console.error(err));
  }, []);

  return (
    <div>
      <h1>Transactions</h1>

      {transactions.map(t => (
        <div key={t.id}>
          <p>{t.title}</p>
        </div>
      ))}
    </div>
  );
}

export default App;
